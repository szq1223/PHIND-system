# image_capture.py

import time
import datetime
import subprocess
import os

class ImageCapture:
    def __init__(self, stop_event, capturing_event):
        self.stop_event = stop_event
        self.capturing_event = capturing_event
        self.image_filenames = []

    def capture_image(self):
        date_string = datetime.datetime.now().strftime('%Y%m%d%H%M%S')
        filename = f"/home/park/images/{date_string}.jpg"
        self.image_filenames.append(filename)
        result = subprocess.run(
            ["libcamera-still", "-o", filename, "--nopreview", "-t", "1", "--width", "640", "--height", "480"],
            capture_output=True,
            text=True
        )
        if result.returncode != 0:
            print(f"Error capturing image: {result.stderr}")
        else:
            print(f"Captured image: {filename}")

    def run(self):
        while not self.stop_event.is_set():
            if self.capturing_event.is_set():
                self.capture_image()
                time.sleep(0.5)  # Capture image every 0.5 seconds
            else:
                time.sleep(0.5)
